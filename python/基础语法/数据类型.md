## 列表(List)

特点:

- 元素有序
- 可扩容
- 元素可异构（支持不同数据类型）



```python
# 1. 创建空列表
l1 = []
l2 = list()

# 2. 字符串转换列表
l3 = list('cat')
print(l3) # ['c', 'a', 't']

# 3. 列表常用操作
data = [1,2,3,4,5,6,7,8,9]
data1 = [0,0,0,0,0,0,0,0]
data2 = [1,2,3,4,5,6,7,8,9]
len(data)                   # 返回序列中元素的个数
max(data)                   # 返回序列中最大值
min(data)                   # 返回序列中最小值
print(data+data1)           # 两个序列拼接
print(data*3)               # 重复序列元素3次
print(6 in data)            # 判断元素是否在序列中,返回True或False

# 4. 列表的索引和切片 与字符串相同
L = ['c','d','e','g','a','f','b','h']
# 语法: L[start:stop:step]
print(L[0])     # c
print(L[3])     # g
print(L[:5])    # ['c', 'd', 'e', 'g', 'a']
print(L[4:])    # ['a', 'f', 'b', 'h']
print(L[2:5])   # ['e', 'g', 'a']
print(L[::-1])  # ['h', 'b', 'f', 'a', 'g', 'e', 'd', 'c']
print(L[::2])   # ['c', 'e', 'a', 'b']


# 5. 向列表末尾增加元素
L = ['a','b','c']
L.append('d')
print(L)  # ['a', 'b', 'c', 'd']

# 6. 对列表中某个元素计数
L = ['a','p','p','l','e']
L.count('p')  # 2

# 7. 列表的拼接
L1 = [1,2]
L2 = ['a','b']
# 方法1：L1.extend(L2)
# 方法2：L3 = L1 + L2

# 8. 查找元素的索引位置
L = ['a','c','c','d']
L.index('c')             # 输出第一个'c'的索引位置

# 9. 列表元素的定向插入
L = ['a', 'b', 'd']
L.insert(2, 'c')                # 在索引为2的位置上插入'c' , 后面的元素后移
L.insert(100, 'e')              # 超出索引也不抛异常 相当于append

# 10. 移出指定索引位置元素， 并返回该值
L = ['a','b','c','d']
L.pop(2)                      # 弹出指定索引的元素， 在这里是 'c'
L.pop()                       # 默认移除最后一个元素

# 11. 列表元素的移除
L = ['a','c','d','a']
L.remove('a')                # 移除第一个 'a' 元素
# 移除不存在的元素报ValueError

# 12. 三种列表反转的方法
L = [1,2,3,4,5,6,7,8,9,0]
L.reverse()                # 第一种  推荐
L[::-1]                    # 第二种
list(reversed(L))          # 第三种

# 13. 列表的排序
L1 = [4,6,2,8,9,5,1,3,7,0]
L2 = ['c','d','e','g','a','f','b','h']
L1.sort()             # 升序
L2.sort(reverse=True) # 降序

# 14. 清空元素
L = [1,2,3,4]
L.clear()
print(L)

# 15. 复制列表
L = [1,2,3]
L2 = L.copy()
L3 = L[:]
L4 = list(L)

# 16. 多维列表
things = [['apple', 'pear'], ['bike', 'bus'], ['teacher', 'worker']]
print(things[1][1])

# 17. 修改列表中元素
fruit = ['apple', 'banana', 'cherry', 'lemon']
fruit[1] = 'orange'
print(fruit)  # ['apple', 'orange', 'cherry', 'lemon']

# 18. 删除指定元素
L = ['a', 'b', 'c', 'd']
del L[2]
print(L)  # ['a', 'b', 'd']

# 19. 判断元素是否存在
L = ['a', 'b', 'c', 'd']
print('a' in L) # True




```





## 字典(Dict)

特点：

- 键值对是无序的
- 通过键来访问值
- 键通常是字符串



```python
# 1. 创建字典两种方式
d1 = {}
d2 = dict()

# 2. 增加一键值对
d3 = {'gender': "male"}
d3["name"] = "Zheng"

# 3. clear() 元素清空
d4 = {"name": "Tom", "age": 12}
print(d4)  # {'name': 'Tom', 'age': 12}
d4.clear()
print(d4)  # {}

# 4. copy() 字典浅拷贝
d5 = {"name": "Tom", "age": 12}
d6 = d5.copy()
print(d6) # {'name': 'Tom', 'age': 12}

# 5. 通过fromkeys()创建字典
# 将seq作为新字典的键, fromkeys的第二参数作为值, 没有第二参数就默认是None
list1 = ['Tom', 'David', 'Bob']

d7 = dict.fromkeys(list1, 10)
print(d7)  # {'Tom': 10, 'David': 10, 'Bob': 10}

# 6. 根据key查对应的值
d8 = {"name": "Tom", "age": 12}

# 通过get()方法 存在就返回对应的值, 不存在就返回get()方法的第二个参数, 默认是None
print(d8.get('name'))          # Tome
print(d8.get('gender'))        # None
print(d8.get("gender", "boy")) # boy

# 通过中括号方式, 存在则返回，不存在则报错(KeyError)
print(d8['name'])
# print(d8['key'])

# 7. 返回一个由元组构成的列表，每个元组包含一对键值对
d9 = {"name": "Tom", "age": 12}
print(d9.items())  # dict_items([('name', 'Tom'), ('age', 12)])

# 8. 返回一个由字典所有键组成的列表
d10 = {"name": "Tom", "age": 12}
print(d10.keys())  # dict_keys(['name', 'age'])

# 9. 返回一个由字典所有值构成的列表
d11 = {"name": "Tom", "age": 12}
print(d11.values())  # dict_values(['Tom', 12])

# 10. pop() 移除指定key的键值对
d12 = {"name": "Tom", "age": 12}
print(d12)
d12.pop('age')
print(d12)  # {'name': 'Tom'}


# 11. popitem() 删除并返回字典中的最后一对键和值
d13 = {"name": "Tom", "age": 12}
print(d13.popitem())  # ('age', 12)
print(d13)  # {'name': 'Tom'}

# 12. 删除键值对
d14 = {"key": "value"}
del d14["key"]
# del d["key"]   # 删除不存在的key, 报KeyError
print(d14)

# 13. setdefault() 如果键不存在于字典中，将会添加键并将键值对
d15 = {"name": "Tom", "age": 12}
d15.setdefault('name')
print(d15)  # {'name': 'Tom', 'age': 12}
d15.setdefault('weight', 50)
print(d15)  # {'name': 'Tom', 'age': 12}


# 14. update() 合并字典
d16 = {"name": "Tom", "age": 12}
d17 = {"name": "Kyle", "age": 13}
d16.update(d17)
print(d16)  # {'name': 'Kyle', 'age': 13}

# 15. 对任何双值序列都可以使用dict()
d18 = [['a', 'b'], ['c', 'd'], ['e', 'f']]
d19 = dict(d18)
print(d19) # {'a': 'b', 'c': 'd', 'e': 'f'}
```





## 元组(Tuple)

- 元组占用的空间较小
- 不能任意修改元组的值
- 可以将元组作为字典的键
- 命名元组可以作为对象的替代

```python

print(t1[1:])       # ('23', [123, 'abc'], ('learn', 'python'))
print(t1[3][1])     # python  多维索引
print(t1[2][0])     # 123


# 创建单元素元组
t2 = (3,)
t2_2 = (3)
print(type(t2))    # <class 'tuple'> 
print(type(t2_2))  # <class 'int'>

# 元组计数 计算元素2出现次数

# 返回某元素第一次出现的索引位置

# 元组拆包
lax_coordinates = (33.9425, -118.408056)
```

## 集合(Set)

- 元素无序
- 元素不重复

```python
# 创建空集合
s0 = {1,2,3,4}
s1 = set()

# add() 元素的添加(位置随机)
s2 = {'c', 'd', 'e'}
s2.add('b')
print(s2)  # {'e', 'b', 'd', 'c'}

# update() 合并集合
s3 = {1, 2}
s4 = {3, 4}
s3.update(s4)
print(s3)  # {1, 2, 3, 4}

# pop() 删除并返回任意set元素
s5 = {1, 2, 3, 4}
print(s5.pop())  # 1

# remove() 删除指定索引处元素，删除不存在元素,则报错(KeyError)
s6 = {1, 2, 3}
s6.remove(2)
print(s6)  # {1, 3}

# discard() 根据值删除指定元素, 删除不存在元素不报错
s7 = {'a', 'b', 'c'}
s7.discard('d')
s7.discard('c')
print(s7)  # {'b', 'a'}


# clear() 清空所有元素
s8 = {'a', 'b', 'c'}
s8.clear()
print(s8)  # set()

# 创建不可修改的集合, 修改就报AttributeError
s9 = frozenset("Python")
print(s9)
# s9.add("guido") 报AttributeError

# 判断元素和集合的关系 包含或不包含
s10 = {'a', 'b', 'c'}
print("h" in s10)         # 返回 False
print("a" in s10)         # 返回 True

# 判断集合之间的关系
s11 = {'a', 'b', 'c', 'd'}
s12 = {'a', 'b'}
print(s11 == s12)            # False
print(s11 != s12)            # True
print(s11 > s12)             # True 说明集合s12是集合s11的子集
print(s12.issubset(s11))     # True 判断s12是否是s11的子集
print(s11.issuperset(s12))   # True 判断s11是否是s12的超集

# 求交集和并集
s13 = {'a', 'b', 'c', 'e', 'f', 'd'}
s14 = {'a', 'b', 'e', 'l', 'o'}

# 两个求并集方法
print(s13 | s14)                          # {'e', 'l', 'b', 'f', 'c', 'a', 'o', 'd'}
print(s13.union(s14))                     # {'e', 'l', 'b', 'f', 'c', 'a', 'o', 'd'}

# 两个求交集方法(求共有的函数)
print(s13 & s14)                          # {'e', 'a', 'b'}
print(s13.intersection(s14))              # {'e', 'a', 'b'}

# 求补集方法
print(s13 - s14)                          # {'c', 'f', 'd'}
print(s13.difference(s14))                # {'c', 'f', 'd'}

print(s13.symmetric_difference(s14))      # 求对称差集 异或集 a-b 和b-a的并集

print(s13 ^ s14)

```
